# Restaurant Management

composer create-project laravel/laravel restaurant-management

# Run Project command

php artisan serve

# migration command

php artisan migrate

# Seeder of Role & User(admin seeder)

php artisan db:seed --class=RoleSeeder
php artisan db:seed --class=UserSeeder
